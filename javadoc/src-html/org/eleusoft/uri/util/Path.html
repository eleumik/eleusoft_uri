010    /**
011     * Path class.
012     * This class depends on the concept of path
013     * as specified in URI RFC 3986,
014     * the path has the following rules resumed
015     * from RFC 3986 Section 3.3:
016     * <ol>
017     * <li>A path is made of segments, separated by 
018     * the slash [/] character.
019     * <li>A path is absolute when it starts with slash [/], 
020     * otherwise is said to be a relative path.
021     * <li>A path can also start with double slash [//] 
022     * <li>When is a relative path, its first 
023     * segment cannot contain a colon [:] character.
024     * <blockquote><code>RFC 3986 4.2 </code>
025     * <p>(..)A path segment that contains a colon character 
026     * (e.g., "<code>this:that</code>")
027     * cannot be used as the first segment of a relative-path reference, as
028     * it would be mistaken for a scheme name.  Such a segment must be
029     * preceded by a dot-segment (e.g., "<code>./this:that</code>") 
030     * to make a relative-path reference.
031     * </blockquote>
032     * <p class=note>For 4.2. In a web application the first segment
033     * of the path is the application context, that should not
034     * contain [:]. Since a web app context name maps to a folder
035     * and windows folders cannot contain <code>\ / ? : * &quot; &lt; | &gt; </code>
036     * so having a [:] in first path is not possible for windows, 
037     * I don't know if is possible in linux.</p>
038     * <p class=note>For 4.2, remember that relative paths cannot be normalized
039     * so there is no risk to remove the <code>./</code> in front 
040     * of the <code>./this:that </code> relative path.
041     * <li>Trimming - Delimiters
042     * <li>From <cite>RFC 3986 Appendix C</cite>
043     * Parsed uris are trimmed and two delimiters are removed:
044     * angle-brackets and quotes.
045     * <blockquote>URIs are often transmitted through formats that do not provide a 
046     * clear context for their interpretation. For example, there are many 
047     * occasions when a URI is included in plain text; examples include 
048     * text sent in email, USENET news, and on printed paper. 
049     * In such cases, it is important to be able to delimit 
050     * the URI from the rest of the text, and in particular from punctuation 
051     * marks that might be mistaken for part of the URI.
052     * In practice, URIs are delimited in a variety of ways, but usually 
053     * within double-quotes "http://example.com/", angle brackets 
054     * <http://example.com/>, or just by using whitespace
055     * <p>....
056     * <p>For robustness, software that accepts user-typed URI should attempt 
057     * to recognize and strip both delimiters and embedded whitespace.</blockquote>
058     * </ol>
059     * <p>A path is not an URI but a component of an URI.
060     * <p>The string representation of a path 
061     * as intended by this class is always unescaped,
062     * the escaped form can be obtained with 
063     * <code>Path.getURI().toString()</code>.
064     * <p><b>Extensions</b>
065     * <p>This class allows to split logically
066     * a path into a 
067     * <em>context</em> and a <em>name</em> part.
068     * The <em>context</em> is what is before the last slash [/] char,
069     * included the slash.
070     * The <em>name</em> is what is after the last slash [/] char,
071     * if there is something.
072     * <p>For example:
073     * <code class='code'>
074     * /dir1/dir2/ciao.xml
075     * context = /dir1/dir2/
076     * name = ciao.xml
077     *
078     * /dir1/dir2/
079     * context = /dir1/dir2/
080     * name = [absent]
081     * </code>.
082     */
083    public final class Path
084    {
085        //private static final Path EMPTY_PATH; // NO empty path..too strange behavior uri impls..
086        
087        /**
088         * A common use Path represening
089         * the parent path [..]
090         */
091        public static final Path PARENT_PATH;
092        /**
093         * A common use Path represening
094         * the context path [.]
095         */
096        public static final Path CONTEXT_PATH;
097        /**
098         * A common use Path represening
099         * the root path [/]
100         */
101        public static final Path ROOT_PATH;
102        
103        
104        static
105        {
106            try
107            {
108                ROOT_PATH = new Path("/");
109                PARENT_PATH = new Path("..");
110                CONTEXT_PATH = new Path(".");
111            }
112            catch(URIException e)
113            {
114                e.printStackTrace();
115                throw new Error("Impossible ? org.eleusoft.uri.util.Path 35324:" + e.getMessage());
116            }
117        }
118        
119        // The contained URI instance.
120        private URI uri;
121        
122        /**
123         * Public contructor.
124         * @throws URIException if the passed path is not a valid path.
125         */
126            public Path(final String path) throws URIException
127            {
128                this(URIFactory.createURI(null, null, path, null, null), true);
129            }
130            
131            /**
132         * Public contructor with URI configuration
133         * @throws URIException if the uri is absolute 
134         * and/or if it has a fragment and/or a query.
135         */
136            public Path(final URI uri) throws URIException
137            {
138                this(uri, true);
139                if (uri.isAbsolute()) throw new URIException("Passed URI is an absolute URI", uri.toString());
140                else if (uri.hasFragment()) throw new URIException("Passed URI has fragment", uri.toString());
141                else if (uri.hasQuery()) throw new URIException("Passed URI has query", uri.toString());
142            }
143            
144            // Fast lane constructor for private impl
145            // called also by other constructors
146            private Path(final URI uri, boolean dummy) throws URIException
147            {
148                this.uri = uri;
149                if (getPath()==null) throw new URIException("Missing path in passed URI:" + uri.toString(),  uri.toString());
150            }
151            /**
152         * Gets the path as an URI interface instance.
153         * <p>Internally retrieves the 
154         * unmodifiable wrapped URI interface.
155         */
156        public URI toURI()
157        {
158            return uri;
159        }
160            /**
161             * Retrieves the string 
162             * value of this path,
163             * in unescaped form;
164             * equivalent to {@link #getPath()}
165             */
166            public String toString()
167            {
168                return getPath();
169            }
170            /**
171             * Retrieves the string 
172             * value of this path,
173             * in unescaped form.
174             */
175            public String getPath()
176            {
177                return uri.getPDPath();
178            }
179            /**
180             * Returns the sequence of path
181             * segments in percent-decoded form
182             * as a String array.
183             * @return a String array, possibly empty, never null.
184             */
185            public String[] getSegments()
186            {
187                final StringTokenizer st = new StringTokenizer(getPath(), "/");
188                final int i = st.countTokens();
189            final String[] s = new String[i];
190            int z = 0;
191            while(st.hasMoreElements())
192            {
193                s[z] = st.nextToken();
194                z ++ ;
195            }
196            return s;
197            }
198            /**
199             * Signal whether this
200             * path is a relative path,
201             * eg: does not start with [/] slash.
202             */
203            public boolean isRelative()
204            {
205                return uri.isRelativePath();
206            }
207            /**
208             * Retrieves the name 
209             * part of this path.
210             * See class description
211             * for description of how 
212             * a path is split in a 
213             * <em>context</em> and a <em>name</em> part.
214             *
215             */
216            public String getName()
217            {
218                return uri.getName();
219            }
220

228             */

230            {
231                return resolve(CONTEXT_PATH);
232                // 201401 old, was disabled:

236    //          //if (true) throw new Error("path:" + path + "] currHier:" + currHier+"[");
237    //          return new Path(currHier);
238            }
239            
240            /**
241             * Retrieves the parent path.
242             * Internally creates a new path obtained
243             * resolving this path using
244             * {@link resolve(Path)} and passing
245             * the {@link PARENT_PATH} instace.
246             */
247            public Path getParentPath() throws URIException
248            {
249                return resolve(PARENT_PATH) ;
250            }
251            /**
252             * Creates a new path obtained
253             * performing URI normalization
254             * on this path.
255             */
256            public Path normalize() throws URIException
257            {
258                URI n = uri.normalize();
259                if (n==uri) return this;
260                else return new Path(n);
261            }
262            
263            /**
264             * Creates a new path obtained
265             * resolving this uri with the
266             * passed <em>relative</em> path.
267             * {@link resolve(Path)} and passing
268             * the {@link PARENT_PATH} instace.
269             * @param relPath a relative path 
270             * @throws URIException when the passed path is not relative.
271             */
272            public Path resolve(Path relPath) throws URIException
273            {
274                // 200706 added check for this.relative to fix test A389 | 279 
275                if(!relPath.isRelative()) throw new URIException(
276                    "Not a relative path:" + relPath, relPath.toString());
277                    else if(isRelative()) throw new URIException(
278                    "This is a relative path [ " + toString() + 
279                    "], cannot resolve with rel path [" + relPath + "]", 
280                    relPath.toString());
281                    
282                    final URI resolved = uri.resolve(relPath.uri).normalize();
283                    // Changed 2000705
284                    //final Path resolvedPath = resolved.getPath()==null ? 
285                    //    CONTEXT_PATH : new Path(resolved, true);
286                    final Path resolvedPath = resolved.isPathEmpty() ? 
287                        ROOT_PATH : new Path(resolved, true);
288                    return resolvedPath;
289            }
290            /**
291             * Convenience method, like calling
292             * {@link resolve(Path)} but passing
293             * a string.
294             * @param relPath a relative path 
295             * @throws URIException when the passed path is not relative.
296             */
297            public Path resolve(String relPath) throws URIException
298            {
299                
300                    if(relPath==null) throw new IllegalArgumentException("null relative path");
301                    
302                    Path relPathObj = new Path(relPath);
303                    return resolve(relPathObj);
304        }
