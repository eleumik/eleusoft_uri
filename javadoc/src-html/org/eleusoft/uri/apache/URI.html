047    /**
048     * The interface for the URI(Uniform Resource Identifiers) version of RFC 2396.
049     * This class has the purpose of supportting of parsing a URI reference to
050     * extend any specific protocols, the character encoding of the protocol to
051     * be transported and the charset of the document.
052     * <p>
053     * A URI is always in an "escaped" form, since escaping or unescaping a
054     * completed URI might change its semantics.
055     * <p>
056     * Implementers should be careful not to escape or unescape the same string
057     * more than once, since unescaping an already unescaped string might lead to
058     * misinterpreting a percent data character as another escaped character,
059     * or vice versa in the case of escaping an already escaped string.
060     * <p>
061     * In order to avoid these problems, data types used as follows:
062     * <p><blockquote><pre>
063     *   URI character sequence: char
064     *   octet sequence: byte
065     *   original character sequence: String
066     * </pre></blockquote><p>
067     *
068     * So, a URI is a sequence of characters as an array of a char type, which
069     * is not always represented as a sequence of octets as an array of byte.
070     * <p>
071     *
072     * URI Syntactic Components
073     * <p><blockquote><pre>
074     * - In general, written as follows:
075     *   Absolute URI = <scheme>:<scheme-specific-part>
076     *   Generic URI = <scheme>://<authority><path>?<query>
077     *
078     * - Syntax
079     *   absoluteURI   = scheme ":" ( hier_part | opaque_part )
080     *   hier_part     = ( net_path | abs_path ) [ "?" query ]
081     *   net_path      = "//" authority [ abs_path ]
082     *   abs_path      = "/"  path_segments
083     * </pre></blockquote><p>
084     *
085     * The following examples illustrate URI that are in common use.
086     * <pre>
087     * ftp://ftp.is.co.za/rfc/rfc1808.txt
088     *    -- ftp scheme for File Transfer Protocol services
089     * gopher://spinaltap.micro.umn.edu/00/Weather/California/Los%20Angeles
090     *    -- gopher scheme for Gopher and Gopher+ Protocol services
091     * http://www.math.uio.no/faq/compression-faq/part1.html
092     *    -- http scheme for Hypertext Transfer Protocol services
093     * mailto:mduerst@ifi.unizh.ch
094     *    -- mailto scheme for electronic mail addresses
095     * news:comp.infosystems.www.servers.unix
096     *    -- news scheme for USENET news groups and articles
097     * telnet://melvyl.ucop.edu/
098     *    -- telnet scheme for interactive services via the TELNET Protocol
099     * </pre>
100     * Please, notice that there are many modifications from URL(RFC 1738) and
101     * relative URL(RFC 1808).
102     * <p>
103     * <b>The expressions for a URI</b>
104     * <p><pre>
105     * For escaped URI forms
106     *  - URI(char[]) // constructor
107     *  - char[] getRawXxx() // method
108     *  - String getEscapedXxx() // method
109     *  - String toString() // method
110     * <p>
111     * For unescaped URI forms
112     *  - URI(String) // constructor
113     *  - String getXXX() // method
114     * </pre><p>
115     *
116     * @author <a href="mailto:jericho@apache.org">Sung-Gu</a>
117     * @author <a href="mailto:mbowler@GargoyleSoftware.com">Mike Bowler</a>
118     * @version $Revision: 1.1 $ $Date: 2002/03/14 15:14:01
119     */

206    
207                    if (path == null) throw new IllegalArgumentException("Null path passed. " +
208                            "\nRFC 3986 3.3 A path is always defined for a URI, " +
209                            "though the defined path may be empty (zero length).");
210    
211                    final char[] schemeChars;
212                    if (scheme!=null) {
213                            schemeChars = validateScheme(scheme);
214                }
215                else schemeChars = null;
216    
217    
218                    // MV: here must first escape the components
219                    // and then pass to parseURIReference(String,true)
220                    // otherwise path with %23 [#] is lost....
221            // NEW: query and frag must be already escaped
222            
223            // validate and contruct the URI character sequence
224            StringBuffer buff = new StringBuffer();
225            if (schemeChars != null) {
226                buff.append(schemeChars);
227                buff.append(':');
228            }
229            if (authority != null) {
230                buff.append("//");
231                buff.append(authority);
232            }
233            if (path.length()!=0) {  // MV DO NOT check for empty path, was:  (path!=null) //accept empty path
234                if ((scheme != null || authority != null)
235                        && isPathARelativePath(path)) {
236                    throw new URIException(URIException.PARSING,
237                            "abs_path requested");
238                } else {
239    
240    
241                                    final String escapedPath = escapePathAsString(path);
242                                    checkNotEmptyEscapedPathAndAuthority(escapedPath, authority);
243                                    buff.append(escapedPath);
244                            }
245    
246    
247            }
248            if (query != null) {
249                buff.append('?');
250                buff.append(query);
251            }
252            if (fragment != null) {
253                buff.append('#');
254                //buff.append(escapeFragmentAsString(fragment));
255                buff.append(fragment); //2007remove
256            }
257            parseUriReference(buff.toString());
258        }

262        {
263            char[] arr = fragment.toCharArray();
264            if (arr.length==0) return EMPTYCHARS;
265            else
266            {
267                if (!validate(arr, URI.ABNF_fragment)) {
268                            throw new URIException(URIException.PARSING, 
269                                "incorrect fragment: [" + fragment + "]");
270                        }
271                        else return arr;
272                
273            }

303            private char[] validateScheme(String scheme) 
304                throws URIException 
305            {
306                    
307                    final char[] schemeChars = scheme.toCharArray();
308                    if (schemeChars.length==0) {
309                            throw new URIException(URIException.PARSING, 
310                                "scheme cannot be an empty string " + 
311                                "(URI cannot start with colon ':').");
312                    }
313                    if (!ALPHA.get(schemeChars[0])) {
314                            throw new URIException(URIException.PARSING, "scheme must start with letter (a-z A-Z) : [" + new String(schemeChars) + "]");
315                    }
316                    else if (!validate(schemeChars, URI.ABNF_scheme)) {
317                            throw new URIException(URIException.PARSING, "incorrect scheme: [" + new String(schemeChars) + "]");
318                    }
319                    else return schemeChars;
320            }

322            private void checkNotEmptyEscapedPathAndAuthority(String path, String auth) throws URIException {
323    
324                    // RFC3986 3.3
325                    // (..) If a URI  does not contain an authority component, then the path cannot begin
326                    // with two slash characters ("//").
327                    // NOTE: empty string is a present, empty authority component
328    
329                    if (auth==null && (path.startsWith("//")))
330                    {
331                            throw new URIException("RFC3986 3.3 if an URI "
332                                    + "does not contain an authority component "
333                                    + "then the path cannot begin with [//]. Escaped path: [" + path + "]");
334                    }
335    
336                    if (isPathARelativePath(path)) {
337                            // RFC 3986 3.3
338                            // If the path is a relative path,
339                            // and so the uri is a relative path reference,
340                            // it cannot contain a colon in its first segment.
341                            // RFC 3986 4.2
342                            // (..)A path segment that contains a colon character (e.g., "this:that")
343                            // cannot be used as the first segment of a relative-path reference, as
344                            // it would be mistaken for a scheme name.  Such a segment must be
345                            // preceded by a dot-segment (e.g., "./this:that") to make a relative-
346                            // path reference.
347                            // -------------
348                            // Yes but after normalization ?
349                            // Relative paths cannot be normalized !!!
350                            // Found a good reason for not normalizing.
351                            final int indxColon = path.indexOf(":");
352                            final int indxSlash = path.indexOf("/");
353                            if (indxColon >= 0 && (indxSlash==-1 || indxColon < indxSlash)) {
354                                    // If there is a colon in the rel path and
355                                    // (the first slash is not present or occurs after the colon),
356                                    // then it MUST start with ./
357                                    if (!path.startsWith("./")) throw new URIException("RFC3986 3.3  a URI reference"
358                                            + " may be a relative-path reference, in which case the "
359                                            + " first path segment cannot contain a colon (\":\") character. Escaped path:" + path);
360    
361                            }
362                    }
363    
364            }

449         * Construct a general URI with the given relative URI.
450         * <p><blockquote><pre>
451         *   URI-reference = [ absoluteURI | relativeURI ] [ "#" fragment ]
452         *   relativeURI   = ( net_path | abs_path | rel_path ) [ "?" query ]
453         * </pre></blockquote><p>
454         * Resolving Relative References to Absolute Form.
455         *
456         * <strong>Examples of Resolving Relative URI References</strong>
457         *
458         * Within an object with a well-defined base URI of
459         * <p><blockquote><pre>
460         *   http://a/b/c/d;p?q
461         * </pre></blockquote><p>
462         * the relative URI would be resolved as follows:
463         *
464         * Normal Examples
465         *
466         * <p><blockquote><pre>
467         *   g:h           =  g:h
468         *   g             =  http://a/b/c/g
469         *   ./g           =  http://a/b/c/g
470         *   g/            =  http://a/b/c/g/
471         *   /g            =  http://a/g
472         *   //g           =  http://g
473         *   ?y            =  http://a/b/c/?y
474         *   g?y           =  http://a/b/c/g?y
475         *   #s            =  (current document)#s
476         *   g#s           =  http://a/b/c/g#s
477         *   g?y#s         =  http://a/b/c/g?y#s
478         *   ;x            =  http://a/b/c/;x
479         *   g;x           =  http://a/b/c/g;x
480         *   g;x?y#s       =  http://a/b/c/g;x?y#s
481         *   .             =  http://a/b/c/
482         *   ./            =  http://a/b/c/
483         *   ..            =  http://a/b/
484         *   ../           =  http://a/b/
485         *   ../g          =  http://a/b/g
486         *   ../..         =  http://a/
487         *   ../../        =  http://a/
488         *   ../../g       =  http://a/g
489         * </pre></blockquote><p>
490         *
491         * Some URI schemes do not allow a hierarchical syntax matching the
492         * <hier_part> syntax, and thus cannot use relative references.
493         *
494         * @param base the base URI
495         * @param relative the relative URI
496         * @throws URIException If the new URI cannot be created.
497         */
498        public URI(URI base, URI relative) throws URIException {
499    
500                    // RFC 3986 5.2.1 says no scheme, no resolving
501                    // TODO add to "strict" option
502            if (base._scheme == null) {
503                throw new URIException(URIException.PARSING, "base URI required");
504            }
505            else if (base._is_opaque_part || relative._is_opaque_part) {
506                            // Fix one is opaque case
507                            // use copyOpaqueAllButScheme
508                            base.copyContext(this);
509                this._scheme = base._scheme;
510                this._is_opaque_part = base._is_opaque_part
511                    || relative._is_opaque_part;
512                this._opaque = relative._opaque; // ??? TODO
513                this._path = this._opaque; // ??? TODO
514                this._fragment = relative._fragment;
515                // Always call setURI before exiting
516                this.setURI();
517                return;
518            }
519    
520                    // Here base is absolute uri,
521                    // none of the two is opaque
522    
523    
524                    // Fix cases relative is absolute and relative
525                    // with authority
526    
527            if (relative._scheme != null) {
528                            // Case relative is absolute URI
529                            // clone the relative apart fragment
530                this._scheme = relative._scheme;
531                relative.copyNotOpaqueAllButScheme(this);
532            } else if (relative._authority != null) {
533                            // Case network path //myhost[..]
534                            // Clone all but scheme and fragment of the base
535                this._scheme = base._scheme;
536                            relative.copyNotOpaqueAllButScheme(this);
537            } else {
538    
539                            // Here relative has no scheme and no authority,
540                            // copy the scheme and authority part from the base to the
541                            // result uri (this for the moment)
542                            this._scheme = base._scheme;
543                            base.copyContext(this);
544    
545                            if (base._authority != null ) {
546                                    base.copyAuthority(this);
547                }
548                // Here remains to solve path, query and fragment.
549                // If the relative has no path bu query and/or fragment
550                // the base path must be used.
551    
552                            if (relative._path.length == 0) {
553                                    // Remember: Resolved fragment, if present, is ALWAYS fragment of relative
554                                    if (relative._query==null) {
555                                            if (relative._fragment==null) {
556                                                    // Relative has no scheme, auth, path, query and fragm
557                                                    // so is the empty uri, clone the base uri query (path done later)
558                                                    this._query = base._query;
559                                            } else {
560                                                    this._fragment = relative._fragment;
561                                                    this._query = base._query;
562                                            }
563                                    } else {
564                                            // Rel (that is rel uri, empty auth/path)
565                                            // has query: copy query and possible fragment
566                                            relative.copyQueryFragment(this);
567                                    }
568                                    base.copyPath(this);
569    
570                            } else {
571                                    // Relative has path, resolve base path with rel
572                                    // and use query and frag from rel
573                                    this._path = resolvePath(base._path, relative._path);
574                                    relative.copyQueryFragment(this);
575    
576                            }
577                    }
578                    this.setURI();
579            }

1772        /**
1773         * Encodes URI string.
1774         *
1775         * This is a two mapping, one from original characters to octets, and
1776         * subsequently a second from octets to URI characters:
1777         * <p><blockquote><pre>
1778         *   original character sequence->octet sequence->URI character sequence
1779         * </pre></blockquote><p>
1780         *
1781         * An escaped octet is encoded as a character triplet, consisting of the
1782         * percent character "%" followed by the two hexadecimal digits
1783         * representing the octet code. For example, "%20" is the escaped
1784         * encoding for the US-ASCII space character.
1785         * <p>
1786         * Conversion from the local filesystem character set to UTF-8 will
1787         * normally involve a two step process. First convert the local character
1788         * set to the UCS; then convert the UCS to UTF-8.
1789         * The first step in the process can be performed by maintaining a mapping
1790         * table that includes the local character set code and the corresponding
1791         * UCS code.
1792         * The next step is to convert the UCS character code to the UTF-8 encoding.
1793         * <p>
1794         * Mapping between vendor codepages can be done in a very similar manner
1795         * as described above.
1796         * <p>
1797         * The only time escape encodings can allowedly be made is when a URI is
1798         * being created from its component parts.  The escape and validate methods
1799         * are internally performed within this method.
1800         *
1801         * @param original the original character sequence
1802         * @param allowed those characters that are allowed within a component
1803         * @param charset the protocol charset
1804         * @return URI character sequence
1805         * @throws URIException null component or unsupported character encoding
1806         */
1807    
1808        protected static char[] encode(String original, BitSet allowed,
1809                String charset) throws URIException    {  
1810           
1811            if (original == null) {
1812                throw new IllegalArgumentException("Original string may not be null");
1813            }
1814            if (allowed == null) {
1815                throw new IllegalArgumentException("Allowed bitset may not be null");
1816            }
1817            byte[] rawdata = URLCodec.encodeUrl(allowed, EncodingUtil.getBytes(original, charset));
1818            //if (MV || true) System.out.println("Encode: [" + original + "] result:[" + EncodingUtil.getAsciiString(rawdata) + "]");
1819            return EncodingUtil.getAsciiString(rawdata).toCharArray();
1820        }

3354         *
3355         * @param path the path to normalize

3358         */
3359        protected static char[] normalize(char[] path) throws URIException {
3360    
3361            //System.out.println("Normalize [" + new String(path) + "]");
3362    
3363            if (path.length==0) {
3364                return path;
3365            }
3366    
3367    
3368    
3369            String normalized = new String(path);
3370    
3371            // If the buffer begins with "./" or "../", the "." or ".." is removed.
3372            if (normalized.startsWith("./")) {
3373                normalized = normalized.substring(1);
3374            } else if (normalized.startsWith("../")) {
3375                normalized = normalized.substring(2);
3376            } else if (normalized.startsWith("..")) {
3377                normalized = normalized.substring(2);
3378            }
3379    
3380            // All occurrences of "/./" in the buffer are replaced with "/"
3381            int index = -1;
3382            while ((index = normalized.indexOf("/./")) != -1) {
3383                normalized = normalized.substring(0, index) + normalized.substring(index + 2);
3384            }
3385    
3386            // If the buffer ends with "/.", the "." is removed.
3387            if (normalized.endsWith("/.")) {
3388                normalized = normalized.substring(0, normalized.length() - 1);
3389            }
3390    
3391            int startIndex = 0;
3392    
3393            // All occurrences of "/<segment>/../" in the buffer, where ".."
3394            // and <segment> are complete path segments, are iteratively replaced
3395            // with "/" in order from left to right until no matching pattern remains.
3396            // If the buffer ends with "/<segment>/..", that is also replaced
3397            // with "/".  Note that <segment> may be empty.
3398            while ((index = normalized.indexOf("/../", startIndex)) != -1) {
3399                int slashIndex = normalized.lastIndexOf('/', index - 1);
3400                if (slashIndex >= 0) {
3401                    normalized = normalized.substring(0, slashIndex) + normalized.substring(index + 3);
3402                } else {
3403                    startIndex = index + 3;
3404                }
