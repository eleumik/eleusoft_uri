018        public URI createURI(String scheme, String auth,
019            String path, String query, String fragment) throws URIException
020        {
021            try
022            {
023                // Double encode and decode because apache uri does not
024                        // understand for #.xml that # is part of the path and
025                        // not fragment also if is passed in the path.
026                        // java.net.URI works good..then return path
027                        // unescaping it..
028                        //URI uri = new URI(null, null, normalize(URLUtil.URLEncode(val, "UTF-8")), null, null);
029    
030                return newURI(scheme, auth, path, query, fragment);
031            }
032            catch(org.eleusoft.uri.apache.URIException use)
033            {
034                StringBuffer sb = new StringBuffer();
035                sb.append("Scheme:").append(scheme).append("\nAuthority:").append(auth);
036                sb.append("\nPath:").append(path).append("\nQuery:").append(auth);
037                sb.append("\nFragment:").append(fragment);
038                String msg = sb.toString();
039                throw new URIException(use.getMessage(), msg, use); //rmvd from msg  + "\nparts:" + msg
040            }
041        }

095            protected URI _normalize() throws URIException
096            {
097                // BASE CLASS does not call this method if this
098                // is an opaque uri.
099    
100    
101                try
102    
103                {
104                    // normalization doesn't change from abs to rel uri
105                    boolean isAbsolute = isAbsolute();
106                    // Fix: when normalizing the path should be null
107                    // if we go over the root.
108                    org.eleusoft.uri.apache.URI normapacheuri =
109                        uri.normalize();
110    
111    
112                    // Fix: when normalizing the path should be null
113                    // if we go over the root.
114                    // Note: cannot test if returns the same instance
115                    // because need to fix special cases...
116                    boolean modified = false;
117                    //if (true) throw new Error();
118                    //if (true) return new ApacheURI(normapacheuri);
119    
120                    String query = normapacheuri.getQuery();
121    
122    
123    
124                    String path = normapacheuri.getPath();
125    
126                    if (path==null)
127                    {
128                        // This is the case of an URI over root
129                        // apache handles in this way,
130                        // so we must use empty string as path, as
131                        // specified in URI javadoc.
132                        if (isStrict())  throwExceptionForStrict("Normalization error, no path. Uri is: [" + toString() + "]");
133                        // 200705 and so why '/' for root, changed.
134                        //path = isAbsolute ? "/" : "";
135                        path = "";
136                        modified = true;
137                    }
138                    else if (path.equals("."))
139                    {
140                        // After normalization remove context points
141                        path = isAbsolute ? "" : "";
142                        modified = true;
143                    }
144                    else if (path.startsWith("/../") || path.startsWith(".."))
145                    {
146                        // This is the case gone over the root, return empty path
147                        // java.net.URI handles in this way the paths over the root..
148                        path = "";
149                        modified = true;
150    
151                    }
152                    else if (path.equals("/."))
153                    {
154                        path = ".";
155                        modified = true;
156                    }
157    
158                    if (path.length()!=0)
159                    {
160                        // If it is a non empty rel path
161                        // make it absolute, normalize and remove front slash.
162                        // This is out of spec. should not normalize relative....TODO setting.
163    
164                        // 6.1. Equvalence
165                        // ...
166                        // In testing for equivalence, applications should not directly
167                        // compare relative references; the references should be converted
168                        // to their respective target URIs before comparison.
169                        // When URIs are compared to select (or avoid) a network action,
170                        // such as retrieval of a representation, fragment components
171                        // (if any) should be excluded from the comparison.
172    
173    
174                        if (!path.startsWith("/"))
175                        {
176                            path = "/" + path;
177                            org.eleusoft.uri.apache.URI temp =
178                                 new org.eleusoft.uri.apache.URI(null, null,
179                                     path , null, null);
180                            temp = temp.normalize();
181                            path = temp.getPath();
182                            int patLen = path.length();
183                            path = patLen==0 ? "" : path.substring(1, patLen);
184                            modified = true;
185                        }
186                    }
187    
188                    if (modified)
189                    {
190                        String fragment =normapacheuri.getEscapedFragment(); 
191                        try
192                        {
193                            return newURI(normapacheuri.getScheme(), 
194                                normapacheuri.getAuthority(),
195                                path, 
196                                query, 
197                                fragment);
198                        }
199                        catch(org.eleusoft.uri.apache.URIException use)
200                        {
201                            throw new IllegalStateException("Cannot apport modifications, scheme:[" + normapacheuri.getScheme()
202                                + "] authority:[" + normapacheuri.getAuthority()
203                                + "] path:[" + path + "] query:[" + query
204                                + "] fragment:[" +  fragment
205                                + "] msg:" + use.getMessage()
206                                + "\nthis uri:" + toString());
207                        }
208                    }
209                    else
210                    {
211                        return new ApacheURI(normapacheuri);
212    
213                    }
214                }
215                catch(org.eleusoft.uri.apache.URIException ue)
216                {
217                    throw new Error("Cannot normalize");//, toString(), ue);
218                }

268                {
269                    throw new URIException("Cannot relativize [" +
270                        toReadableURI() + "] with rel [" + relative.toReadableURI() +
271                        "]\nMsg:" + use.getMessage(), relative.toString(), use);
272                }

300        private static URI newURI(String escaped) throws URIException
301        {
302            try
303            {
304                final org.eleusoft.uri.apache.URI impl;
305                // TODO200705 FINDBUGS not used isOpaque and getOpaqueParts..?
306                if (true || !isOpaque(escaped))
307                {
308                    // END - impl =  new org.eleusoft.uri.apache.URI (escaped, true);
309                    impl = new org.eleusoft.uri.apache.URI (escaped);
310                }
311                else
312                {
313                    String[] parts = getOpaqueParts(escaped);
314                    impl = new org.eleusoft.uri.apache.URI (parts[0], parts[1], parts[2]);
315                    if (!impl.isOpaquePart()) throw new Error("caazzzz");
316                }
317                final ApacheURI result = new ApacheURI(impl);
318                //if (!escaped.equals(impl.getEscapedURIReference()))
319                // throw new RuntimeException("result is :" + impl.getEscapedURIReference() + " while source:" + escaped);
320                return result;
321            }
322            catch(org.eleusoft.uri.apache.URIException use)
323            {
324                throw new URIException(use.getMessage(), escaped, use);
325            }
326        }

328        /**
329         * @return true if passed abs uri is opaque
330         * @param uri an ESCAPED  absolute uri to be tested for opacity
331         */
332        private static boolean isOpaque(String uri) throws URIException
333        {
334            //System.out.println("Call is opaque:" + uri);
335            int i = uri.indexOf(':');
336            if (i==-1) return false;
337            else if (i==0) throw new URIException("First character is a colon. If it is a relative path should be escaped, otherwise is an error", uri);
338            else if (uri.length() <= i + 2)
339            {
340                // here can trap http:. and http:..
341                //System.out.println("Not opaque, Exit for:" + uri);
342                return false; // nothing or less than 2 chars after : of scheme
343            }
344            else
345            {
346                if (uri.substring(++i, i+1).equals("/"))
347                {
348                    System.out.println("Not opaque");
349                    return false;
350                }
351                else
352                {
353                    //System.out.println("Is opaque, Non / but:" + uri.substring(i, i+2));
354                    return true;
355                }
356            }
357        }

359        /**
360         * @return true if passed abs uri is opaque
361         * @param uri an ESCAPED absolute uri to be tested for opacity
362         */
363        private static String[] getOpaqueParts(String uri)
364        {
365            //System.out.println("Call getOpaqueParts:" + uri);
366            int i = uri.indexOf(':');
367            if (i==-1) throw new IllegalStateException("Is opaque and there is no ':' character ?");
368    
369            final String scheme = uri.substring(0, i);
370            // Scheme specific part
371            final String ssp;
372            final String fragment;
373            // increment to skip [:]
374            ++i;
375           int len = uri.length();
376            int h = uri.lastIndexOf('#');
377            if (h==-1)
378            {
379                ssp = uri.substring(i, len);
380                fragment = null;
381            }
382            else
383            {
384                ssp = uri.substring(i,h);
385                // increment to skip [#]
386                h++;
387                fragment = (h>len) ? "" : uri.substring(h, len);
388            }
389            //System.out.println("From uri:" + uri + "\nparts:" + scheme + "]\n" + ssp + "]\n" + fragment + "]");
390            return new String[]{scheme, ssp, fragment};
391        }

393     // P-prot for faster access from inner class.
394        static String buildURIString(String scheme, 
395                String authority, 
396                String path, 
397                String query, 
398                String fragment)
399        {
400            final StringBuffer sb = new StringBuffer();
401            if (scheme!=null) sb.append(scheme).append(':');
402            if (authority!=null) sb.append("//").append(authority);
403            if (path!=null) sb.append(path);
404            if (query!=null) sb.append('?').append(query);
405            if (fragment!=null) sb.append('#').append(fragment);
406            return sb.toString();
407        }
