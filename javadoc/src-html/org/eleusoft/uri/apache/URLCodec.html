035    /**
036     * <p>Implements the 'www-form-urlencoded' encoding scheme,
037     * also misleadingly known as URL encoding.</p>
038     *
039     * <p>For more detailed information please refer to
040     * <a href="http://www.w3.org/TR/html4/interact/forms.html#h-17.13.4.1">
041     * Chapter 17.13.4 'Form content types'</a> of the
042     * <a href="http://www.w3.org/TR/html4/">HTML 4.01 Specification<a></p>
043     *
044     * <p>
045     * This codec is meant to be a replacement for standard Java classes
046     * {@link java.net.URLEncoder} and {@link java.net.URLDecoder}
047     * on older Java platforms, as these classes in Java versions below
048     * 1.4 rely on the platform's default charset encoding.
049     * </p>
050     *
051     * @author Apache Software Foundation
052     * @since 1.2
053     * @version $Id: URLCodec.java,v 1.1 2007/07/10 18:19:30 mik Exp $
054     */
055    public class URLCodec //implements BinaryEncoder, BinaryDecoder, StringEncoder, StringDecoder
056    {
057    
058        /**
059         * The default charset used for string decoding and encoding.
060         */
061        protected String charset = "UTF-8";
062        
063        private static final String US_ASCII = "US-ASCII";
064    
065        protected static final byte ESCAPE_CHAR = '%';
066        /**
067         * BitSet of www-form-url safe characters.
068         */
069        protected static final BitSet WWW_FORM_URL = new BitSet(256);
070    
071        // Static initializer for www_form_url
072        static {
073            // alpha characters
074            for (int i = 'a'; i <= 'z'; i++) {
075                WWW_FORM_URL.set(i);
076            }
077            for (int i = 'A'; i <= 'Z'; i++) {
078                WWW_FORM_URL.set(i);
079            }
080            // numeric characters
081            for (int i = '0'; i <= '9'; i++) {
082                WWW_FORM_URL.set(i);
083            }
084            // special chars
085            WWW_FORM_URL.set('-');
086            WWW_FORM_URL.set('_');
087            WWW_FORM_URL.set('.');
088            WWW_FORM_URL.set('*');
089            // blank to be replaced with +
090            WWW_FORM_URL.set(' ');
091        }
092    
093    
094        /**
095         * Default constructor.
096         */
097        public URLCodec() {
098            super();
099        }
100    
101        /**
102         * Constructor which allows for the selection of a default charset
103         *
104         * @param charset the default string charset to use.
105         */
106        public URLCodec(String charset) {
107            super();
108            this.charset = charset;
109        }
110    
111        /**
112         * Encodes an array of bytes into an array of URL safe 7-bit
113         * characters. Unsafe characters are escaped.
114         *
115         * @param urlsafe bitset of characters deemed URL safe
116         * @param bytes array of bytes to convert to URL safe characters
117         * @return array of bytes containing URL safe characters
118         */
119        public static final byte[] encodeUrl(BitSet urlsafe, byte[] bytes)
120        {
121            if (bytes == null) {
122                return null;
123            }
124            if (urlsafe == null) {
125                urlsafe = WWW_FORM_URL;
126            }
127    
128            ByteArrayOutputStream buffer = new ByteArrayOutputStream();
129            for (int i = 0; i < bytes.length; i++) {
130                int b = bytes[i];
131                if (b < 0) {
132                    b = 256 + b;
133                }
134                if (urlsafe.get(b)) {
135                    if (b == ' ') {
136                        b = '+';
137                    }
138                    buffer.write(b);
139                } else {
140                    buffer.write('%');
141                    char hex1 = Character.toUpperCase(
142                      Character.forDigit((b >> 4) & 0xF, 16));
143                    char hex2 = Character.toUpperCase(
144                      Character.forDigit(b & 0xF, 16));
145                    buffer.write(hex1);
146                    buffer.write(hex2);
147                }
148            }
149            return buffer.toByteArray();
150        }
151    
152    
153        /**
154         * Decodes an array of URL safe 7-bit characters into an array of
155         * original bytes. Escaped characters are converted back to their
156         * original representation.
157         *
158         * @param bytes array of URL safe characters
159         * @return array of original bytes
160         * @throws CodecException Thrown if URL decoding is unsuccessful
161         */
162        public static final byte[] decodeUrl(byte[] bytes)
163             throws CodecException
164        {
165            if (bytes == null) {
166                return null;
167            }
168            ByteArrayOutputStream buffer = new ByteArrayOutputStream();
169            for (int i = 0; i < bytes.length; i++) {
170                int b = bytes[i];
171                if (b == '+') {
172                    buffer.write(' ');
173                } else if (b == '%') {
174                    try {
175                        int u = Character.digit((char)bytes[++i], 16);
176                        int l = Character.digit((char)bytes[++i], 16);
177                        if (u == -1 || l == -1) {
178                            throw new CodecException("Invalid URL encoding");
179                        }
180                        buffer.write((char)((u << 4) + l));
181                    } catch(ArrayIndexOutOfBoundsException e) {
182                        throw new CodecException("Invalid URL encoding");
183                    }
184                } else {
185                    buffer.write(b);
186                }
187            }
188            return buffer.toByteArray();
189        }
190    
191    
192        /**
193         * Encodes an array of bytes into an array of URL safe 7-bit
194         * characters. Unsafe characters are escaped.
195         *
196         * @param bytes array of bytes to convert to URL safe characters
197         * @return array of bytes containing URL safe characters
198         */
199        public byte[] encode(byte[] bytes) {
200            return encodeUrl(WWW_FORM_URL, bytes);
201        }
202    
203    
204        /**
205         * Decodes an array of URL safe 7-bit characters into an array of
206         * original bytes. Escaped characters are converted back to their
207         * original representation.
208         *
209         * @param bytes array of URL safe characters
210         * @return array of original bytes
211         * @throws CodecException Thrown if URL decoding is unsuccessful
212         */
213        public byte[] decode(byte[] bytes) throws CodecException {
214            return decodeUrl(bytes);
215        }
216    
217    
218        /**
219         * Encodes a string into its URL safe form using the specified
220         * string charset. Unsafe characters are escaped.
221         *
222         * @param pString string to convert to a URL safe form
223         * @param charset the charset for pString
224         * @return URL safe string
225         * @throws UnsupportedEncodingException Thrown if charset is not
226         *                                      supported
227         */
228        public String encode(String pString, String charset)
229            throws UnsupportedEncodingException
230        {
231            if (pString == null) {
232                return null;
233            }
234            return new String(encode(pString.getBytes(charset)), US_ASCII);//StringEncodings.US_ASCII
235        }
236    
237    
238        /**
239         * Encodes a string into its URL safe form using the default string
240         * charset. Unsafe characters are escaped.
241         *
242         * @param pString string to convert to a URL safe form
243         * @return URL safe string
244         * @throws EncoderException Thrown if URL encoding is unsuccessful
245         *
246         * @see #getDefaultCharset()
247         */
248        public String encode(String pString) 
249            throws CodecException, UnsupportedEncodingException {
250            if (pString == null) {
251                return null;
252            }
253            return encode(pString, getDefaultCharset());
254            
255        }
256    
257    
258        /**
259         * Decodes a URL safe string into its original form using the
260         * specified encoding. Escaped characters are converted back
261         * to their original representation.
262         *
263         * @param pString URL safe string to convert into its original form
264         * @param charset the original string charset
265         * @return original string
266         * @throws CodecException Thrown if URL decoding is unsuccessful
267         * @throws UnsupportedEncodingException Thrown if charset is not
268         *                                      supported
269         */
270        public String decode(String pString, String charset)
271            throws CodecException, UnsupportedEncodingException
272        {
273            if (pString == null) {
274                return null;
275            }
276            return new String(decode(pString.getBytes(US_ASCII)), charset);
277        }
278    
279    
280        /**
281         * Decodes a URL safe string into its original form using the default
282         * string charset. Escaped characters are converted back to their
283         * original representation.
284         *
285         * @param pString URL safe string to convert into its original form
286         * @return original string
287         * @throws CodecException Thrown if URL decoding is unsuccessful
288         *
289         * @see #getDefaultCharset()
290         */
291        public String decode(String pString) throws CodecException {
292            if (pString == null) {
293                return null;
294            }
295            try {
296                return decode(pString, getDefaultCharset());
297            } catch(UnsupportedEncodingException e) {
298                throw new CodecException(e.getMessage());
299            }
300        }
301    
302        
303        /**
304         * The <code>String</code> encoding used for decoding and encoding.
305         *
306         * @return Returns the encoding.
307         *
308         * @deprecated use #getDefaultCharset()
309         */
310        public String getEncoding() {
311            return this.charset;
312        }
313    
314        /**
315         * The default charset used for string decoding and encoding.
316         *
317         * @return the default string charset.
318         */
319        public String getDefaultCharset() {
320            return this.charset;
321        }
