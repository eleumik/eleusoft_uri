029    public class BrowserURIFactory
030    {
031            private BrowserURIFactory(){}
032    
033            public static URI createURI(String browserURI)
034                    throws URIException
035            {
036                    return URIFactory.createURI(interpretURI(browserURI));
037            }
038            public static String interpretURI(String browserURI)
039            {
040                    StringBuffer sb = new StringBuffer();
041                    char[] arr = browserURI.trim().toCharArray();
042                    for(int i=0,l=arr.length;i<l;i++)
043                    {
044                            char c = arr[i];
045                            if (c!=10 && c!=13 && c!=9)
046                            {
047                                    if (c==32)
048                                {
049                                    sb.append("%20");
050                                }
051                                else sb.append(c);
052                            }
053                    }
054                    return sb.toString();
055            }
