033        private URI __createURI2(String scheme,
034            String auth,
035            String path,
036            String query,
037            String fragment) throws URIException
038        {
039            try
040            {
041                return createURI(buildURIString(scheme,
042                    auth,
043                    new java.net.URI(null, null, path, null, null).toString(),
044                    query,
045                    fragment));
046            }
047            catch (java.net.URISyntaxException use)
048            {
049                StringBuffer sb = new StringBuffer();
050                sb.append("\nScheme:")
051                    .append(scheme)
052                    .append("\nAuthority:")
053                    .append(auth);
054                sb.append("\nPath:").append(path).append("\nQuery:").append(query);
055                sb.append("\nFragment:").append(fragment);
056                String msg = sb.toString();
057                throw new URIException(use.getMessage(), msg, use);
058            }
059        }

061        public URI createURI(String scheme,
062            String auth,
063            String path,
064            String query,
065            String fragment) throws URIException
066        {
067            // RFC3986 3.3
068            // (..) If a URI does not contain an authority component,
069            // then the [path component] cannot begin
070            // with two slash characters ("//").
071    
072            if (auth == null && (path != null && path.startsWith("//")))
073            {
074                throw new URIException("RFC3986 3.3 if an URI " + "does not contain an authority component then "
075                    + "the path cannot begin with [//]",
076                    path);
077            }
078            else
079            {
080                if (true) // This way has 4 errors less
081                    return __createURI2(scheme, auth, path, query, fragment);
082    
083                try
084                {
085                    return new JDK14URI(new java.net.URI(scheme,
086                        auth,
087                        path,
088                        query,
089                        fragment));
090                }
091                catch (java.net.URISyntaxException use)
092                {
093                    StringBuffer sb = new StringBuffer();
094                    sb.append("\nScheme:")
095                        .append(scheme)
096                        .append("\nAuthority:")
097                        .append(auth);
098                    sb.append("\nPath:")
099                        .append(path)
100                        .append("\nQuery:")
101                        .append(query);
102                    sb.append("\nFragment:").append(fragment);
103                    String msg = sb.toString();
104                    throw new URIException(use.getMessage(), msg, use);
105                }
106            }
107        }

109        private static class JDK14URI extends AbstractJDKURI
110        {
111    
112            JDK14URI(String escaped) throws java.net.URISyntaxException,
113                URIException
114            {
115                super(new java.net.URI(removeContextDelimiters(escaped)));
116            }
117    
118            JDK14URI(java.net.URI uri) throws URIException
119            {
120                super(uri);
121            }
122    
123            protected URI _relativize(URI passed) throws URIException
124            {
125                java.net.URI relativized = this.uri.relativize(((JDK14URI) passed).uri);
126                return new JDK14URI(relativized);
127            }
128    
129            protected URI _normalize() throws URIException
130            {
131                // BASE CLASS does not call this method if this
132                // is an opaque uri.
133    
134                // Fix: when normalizing the path should be null
135                // if we go over the root.
136                // Note: cannot test if returns the same instance
137                // because need to fix special cases...
138                boolean modified = false;
139                java.net.URI normjuri = uri.normalize();
140                if (true) return new JDK14URI(normjuri);
141    
142                String query = normjuri.getQuery();
143                ;
144                String path = normjuri.getRawPath();
145    
146                if (path == null)
147                {
148                    // This is the case of an opaque uri
149                    // but for opaque uri this method
150                    // should not have been called.
151                    throw new Error();
152                }
153                else if (path.startsWith("/..") || path.startsWith("../")
154                    || path.equals(".."))
155                {
156                    // This is the case gone over the root, throw error or use an
157                    // empty path
158                    // java.net.URI handles in this way the paths over the root..
159                    if (isStrict())
160                        throwExceptionForStrict("Cannot normalize a path over the root: [" + path
161                            + "]");
162                    path = "";
163                    modified = true;
164    
165                }
166                // if ((path.length()==0 || path.equals("."))
167                else if (path.equals("/."))
168                {
169                    path = "/";
170                    modified = true;
171    
172                }
173    
174                if (modified)
175                {
176                    try
177                    {
178                        /**
179                         * return new JDK14URI( new
180                         * java.net.URI(normjuri.getScheme(),
181                         * normjuri.getAuthority(), path, query,
182                         * normjuri.getFragment()));
183                         **/
184                        return new JDK14URI(buildURIString(normjuri.getScheme(),
185                            normjuri.getAuthority(),
186                            path,
187                            query,
188                            normjuri.getRawFragment()));
189                    }
190                    catch (java.net.URISyntaxException use)
191                    {
192                        throw new IllegalStateException("Implementation error. Cannot apport modifications, path:[" + path
193                            + "] query:["
194                            + query
195                            + "]");
196                    }
197                }
198                else
199                {
200                    return new JDK14URI(normjuri);
201                }
202            }

204            protected URI _resolve(URI relative) throws URIException
205            {
206                // 200705 Fixes test A285 | 175
207                if (isRelativePath())
208                    throw new URIException("This is a relative-path URI [ " + toString()
209                        + "], cannot resolve with rel URI ["
210                        + relative
211                        + "]",
212                        toString());
213    
214                try
215                {
216                    // /////////// FROM HERE FIRST PART IS THE SAME IN APACHE URI
217    
218                    // Baseclass does not call this method
219                    // when relative is absolute or this is opaque.
220                    // if (relative.isAbsolute()) return relative;
221    
222                    // Always only the frag of the rel
223                    final String relativeURIFragment = relative.getPEFragment();
224    
225                    // Fix for 7 ?y http://a/b/c/d;p?y
226                    // Java uri removes the name part
227                    // and outputs http://a/b/c/?y
228                    // Was checking for null..
229                    // now use isPathEmpty()
230                    // String relPath = relative.getPEPath();
231                    if (relative.isPathEmpty() && relative.getAuthority() == null)
232                    {
233    
234                        // Relative URI
235                        // No path
236                        // --> Can have authority, query or fragment
237                        // Do not check for empty query here,
238                        // only on normalize.
239    
240                        final String relURIAuth = relative.getAuthority();
241                        final String relURIQuery = relative.getQuery();
242    
243                        final String toUSEAuth = relURIAuth == null ? getAuthority()
244                            : relURIAuth;
245                        final String toUSEQuery;
246                        if (relURIQuery == null)
247                        {
248                            if (relURIAuth == null && relativeURIFragment == null)
249                            {
250                                // Rel URI, no auth, No path, no query, no fragment.
251                                // Is an empty string. Nothing to do apart removing
252                                // the fragment
253                                return this.removeFragment();
254                            }
255                            else
256                            {
257                                toUSEQuery = this.getQuery();
258                            }
259    
260                        }
261                        else
262                        {
263                            toUSEQuery = relURIQuery;
264                        }
265    
266                        // Return this absolute uri with authority, query and
267                        // fragment
268                        // as determined
269                        // System.out.println("NO AUTH ON REL RESOLVED PATH :" +
270                        // getPath());
271                        // TODO : Return path "/" if this is absolute and has no
272                        // path ??
273                        return new JDK14URI(new java.net.URI(buildURIString(getScheme(),
274                            toUSEAuth,
275                            getPEPath(),
276                            toUSEQuery,
277                            relativeURIFragment)));
278    
279                    }
280                    else
281                    {
282                        // FIX tragical error..
283                        // <http://www.example.org> + <myrelpat>
284                        // is resolved as
285                        // http://www.example.orgmyrelpat
286                        java.net.URI toBeUsedBaseURI;
287                        if (isAbsolute() && getPath().length() == 0)
288                        {
289                            toBeUsedBaseURI = new java.net.URI(buildURIString(getScheme(),
290                                getAuthority(),
291                                "/",
292                                getQuery(),
293                                getPEFragment()));
294                        }
295                        else
296                        {
297                            toBeUsedBaseURI = this.uri;
298                        }
299                        // Case 2: Path is present in relative uri
300                        final java.net.URI resolved = toBeUsedBaseURI.resolve(new java.net.URI(relative.toString()));
301                        String path = resolved.getPath();
302                        // For "normal" test 15 "" http://a/b/c/d;p?q
303    
304                        // efficiency only test
305                        // ?? 200705 ?? what ??
306                        if (path.indexOf("/.") != -1)
307                        {
308                            path = removePathDots(path);
309    
310                            // return
311    
312                            // if (path==null || || path.startsWith(".."))
313                            {
314                                return new JDK14URI(buildURIString(resolved.getScheme(),
315                                    resolved.getAuthority(),
316                                    new java.net.URI(null, null, path, null).toString(),
317                                    resolved.getRawQuery(),
318                                    relativeURIFragment));
319                                // throw new
320                                // URIException("Path error, over the root:" + path,
321                                // path);
322                            }
323    
324                        }
325                        return new JDK14URI(resolved);
326                    }
327                }
328                catch (java.net.URISyntaxException use)
329                {
330                    throw new URIException("Cannot relativize:" + relative.toReadableURI(),
331                        relative.toString(),
332                        use);
333                }
334            }

336            public final URI removeFragment() throws URIException
337            {
338                try
339                {
340                    return !hasFragment() ? this
341                        : new JDK14URI(new java.net.URI(buildURIString(uri.getScheme(),
342                            uri.getAuthority(),
343                            uri.getRawPath(),
344                            uri.getRawQuery(),
345                            null)));
346                }
347                catch (java.net.URISyntaxException use)
348                {
349                    // TODO hmm..very strange if exception..should remove
350                    // URI exception from interface.
351                    throw new URIException("Cannot remove fragment:" + toReadableURI(),
352                        toString(),
353                        use);
354                }
355            }

359        // P-prot for faster access from inner class.
360        static String buildURIString(String scheme,
361            String authority,
362            String path,
363            String query,
364            String fragment)
365        {
366            final StringBuffer sb = new StringBuffer();
367            if (scheme != null) sb.append(scheme).append(':');
368            if (authority != null) sb.append("//").append(authority);
369            if (path != null) sb.append(path);
370            if (query != null) sb.append('?').append(query);
371            if (fragment != null) sb.append('#').append(fragment);
372            return sb.toString();
373        }
