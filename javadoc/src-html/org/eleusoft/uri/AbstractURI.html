034        protected static String removeContextDelimiters(String uri) throws URIException
035        {
036            if (uri==null) throw new IllegalArgumentException("Null uri passed");
037            uri = uri.trim();
038            if (uri.length()>2)
039            {
040                char first = uri.charAt(0);
041                if (isOpenContextDelimiter(first))
042                {
043                    final int ilast = uri.length() - 1;
044                    char clast = uri.charAt(ilast);
045                    if (!isCloseContextDelimiter(clast, first))
046                        throw new URIException("First character is a context delimiter [" + 
047                            first + "] but last not: [" + clast + "]", uri);
048                    return uri.substring(1, ilast); 
049                }    
050            }
051            return uri;
052        } 
053        private static final boolean isOpenContextDelimiter(char c)
054        {
055            return c=='"' || c=='<';
056        }
057        private static final boolean isCloseContextDelimiter(char c, char o)
058        {
059            if (o=='"') return c == '"';
060            else return  c=='>';
061        }

237         * Returns whether the passed URI instances
238         * have the same scheme, as follows:
239         * <ol>
240         * <li>If one of the uri has <em>absent</em> scheme also the 
241         * other must have <em>absent</em> scheme.
242         * <li>Otherwise the scheme must be equal ignoring the
243         * case.
244         * </ol>

246        protected final boolean schemeEquals(URI one, URI two)
247        {
248            final String scheme1 = one.getScheme();
249            final String scheme2 = two.getScheme();
250            if (scheme1==null) return scheme2==null;
251            else return scheme1.equalsIgnoreCase(scheme2);

367        /**
368         * Test an object if this URI is equal to another.
369         *
370         * @param obj an object to compare
371         * @return true if two URI objects are equal
372         */
373        public boolean equals(Object obj) {
374    
375            // normalize and test each components
376            if (obj == this) return true;
377            else if (!(obj instanceof URI))  return false;
378            
379            URI another = (URI) obj;
380            // scheme
381            if (!equals(getScheme(), another.getScheme())) {
382               // System.out.println("DIFF SKEME");
383                return false;
384            }
385            // is_opaque_part or is_hier_part?  and opaque
386            if (isOpaque() != another.isOpaque()) {
387               // System.out.println("DIFF OPAQUE");
388                return false;
389            }
390            // is_hier_part
391            // has_authority
392            if (!equals(getAuthority(), another.getAuthority())) {
393               // System.out.println("DIFF AUTH");
394                return false;
395            }
396            // path
397            if (!equals(getPEPath(), another.getPEPath())) {
398              // System.out.println("DIFF PATH");
399                 return false;
400            }
401            // has_query
402            if (!equals(getQuery(), another.getQuery())) {
403              //  System.out.println("DIFF QUERY");
404                return false;
405            }
406            // has_fragment?  should be careful of the only fragment case.
407            if (!equals(getFragment(), another.getFragment())) {
408              //  System.out.println("DIFF FRAG");
409                return false;
410            }
411            return true;
412        }

431        /**
432         * Compare this URI to another object.
433         *
434         * @param obj the object to be compared.
435         * @return 0, if it's same,
436         * -1, if failed, first being compared with in the authority component
437         * @throws ClassCastException not URI argument
438         */
439        public int compareTo(Object obj) throws ClassCastException {
440    
441            URI another = (URI) obj;
442            if (!equals(getAuthority(), another.getAuthority())) {
443                return -1;
444            }
445            return toString().compareTo(another.toString());
