001    package org.eleusoft.uri;
002    
003    /**
004     * <em>Providers Only</em>
005     * @author mik
006     *
007     */
008    public interface URICodec
009    {
010        /**
011         * Decodes an percent-escaped string following URI rules and using the
012         * passed encoding.
013         * <p>
014         * Note: <code>+</code> remains <code>+</code>, while following URL rules it
015         * becomes a space.
016         * 
017         * @throws URIException when
018         *             <ul>
019         *             <li>The syntax of the escaped uri is wrong.
020         *             </ul>
021         * @return the decoded string, never <code>null</code>
022         */
023        public String decode(final String encoded,
024            final String encoding) throws URIException;
025        
026        /**
027         * Percent-escapes a string following rules for single param name or param
028         * value of a URI query component and using the passed encoding.
029         * 
030         * @return the percent-escaped string, never <code>null</code>
031         */
032        public String encode(final String decoded,
033            final String encoding);
034    
035        
036    
037    }
