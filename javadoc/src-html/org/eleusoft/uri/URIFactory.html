129        public final static boolean isAbsoluteURI(final String uri)
130            throws URIException
131        {
132            return parseScheme(uri)[0]!=null;
133        }
134        /**
135         * Parses an uri extracting the scheme when present.
136         * This method returns an array of two strings.
137         * When there is a scheme the scheme is returned
138         * as first item of the array and
139         * the remaining or whole part is in the second item.
140         * <p>When there is no scheme the first item in the
141         * array is null and the second contains the passed uri.
142         * <p>This method does not check if the scheme
143         * specific part respects the generic uri syntax rules.
144         * @throws URIException when the passed uri starts with
145         *  colon char ':'
146         */
147        public final static String[] parseScheme(final String uri)
148            throws URIException
149        {
150            // scheme: ALPHA *( ALPHA / DIGIT / "+" / "-" / "." )
151            if (uri.length()==0) return new String[]{null, uri};
152            for(int i=0,l=uri.length();i<l;i++)
153            {
154                final char c = uri.charAt(i);
155                if (c==':')
156                {
157                    if (i==0) throw new URIException(
158                        "uri cannot start with ':'",
159                        uri);
160                    return new String[]{
161                        uri.substring(0, i),
162                        uri.substring(i+1)};
163                }
164                else if (!isSchemeChar(c))
165                {
166                    return new String[]{
167                        null,
168                        uri};
169                }
170            }
171            return new String[]{
172                        null,
173                        uri};
174    
175        }
176        /**
177         * Parses an uri extracting the fragment when present.
178         * This method returns an array of two strings.
179         * <ol>
180         * <li>The first item of the array contains the passed
181         * URI without the fragment component, that means
182         * all the string content that preceeds the first '#' symbol
183         * or all the content if there is no '#'.
184         * that is not a fragment. The first item will never be null.
185         * <li>The second item of the array
186         * contains the fragment component, if present,
187         * otherwise null; when the fragment is present
188         * it might be also a zero length string.
189         * <p>This method does not check if the URI
190         * specific part respects the generic uri syntax rules.
191         */
192        public final static String[] extractFragment(final String uri)
193        {
194            if (uri.length()==0) return new String[]{"", null}; // static value
195    
196            for(int i=0,l=uri.length();i<l;i++)
197            {
198                final char c = uri.charAt(i);
199                if (c=='#')
200                {
201                    if (i==0)
202                        return new String[]{
203                            "",
204                            uri.substring(1)}; // 1 == remove #
205                    else
206                        return new String[]{
207                            uri.substring(0, i),
208                            uri.substring(i+1)}; // +1 == remove #
209    
210                }
211    
212            }
213            return new String[]{
214                        uri,
215                        null}; // no frag
216    
217        }

219        /**
220         * Returns the scheme and opaque part of an opaque URI,
221         * null if the URI is not opaque..MAY BE NO MUCH SENSE IN THIS..
222         */
223        public final static String[] extractOpaquePart(final String uri)
224            throws URIException
225        {
226            final String[] parts = parseScheme(uri);
227            final String scheme = parts[0];
228            final String schemeSpecific = parts[1];
229            if (scheme!=null &&
230                (schemeSpecific.length()==0 || schemeSpecific.charAt(0)!='/'))
231            {
232                return new String[]{scheme, schemeSpecific};
233            }
234            else
235            {
236                return null;
237            }
238        }

241        private static boolean isSchemeChar(char c)
242        {
243            return (c >='a' && c <= 'z') ||
244                (c >= 'A' && c <= 'Z') ||
245                 c=='+' ||
246                 c=='-' ||
247                 c=='.';
248        }
249        /**
250         * Same as {@link createURI(String)} but throws
251         * IllegalArgumentException. It's use is intended
252         * for static constants.
253         * @throws IllegalArgumentException when calling
254         * the {@link createURI(String)} method passing
255         * the same uri as parameter throws URIException.
256         */
257        public final static URI uncheckedCreateURI(
258            final String uri) throws IllegalArgumentException
259        {
260            try
261            {
262                return createURI(uri);
263            }
264            catch(URIException e)
265            {
266                e.printStackTrace();
267                throw new IllegalArgumentException(
268                    "Could not create uri [" + uri + "] Msg:" + e.getMessage());
269            }
270        }

314        /**
315         * Creates an {@link URI} instance
316         * from the components of an hierarchical uri .
317         *
318         * @param scheme the scheme component, might be null or not empty.
319         *  Passing an empty string as scheme is an error and will cause an IllegalArgumentException.
320         * @param auth the authority component, might be null or be non empty.
321         *  Passing an empty authority must be interpreted by implementation
322         *  as a null authority.
323         * @param path the unescaped path
324         *  If a URI contains an authority component,
325         *  then the path component must either be empty
326         *  or begin with a slash ("/") character. Cannot be null.
327         * @param query the escaped query
328         * @param fragment the un-escaped fragment
329         *
330         * @throws URIException when one of the components is
331         * wrong for its syntax or for the context, for example
332         * when the path is null or when
333         *  <ul>
334         *  <li>Scheme is an empty string.
335         *  <li>Authority is an empty string.
336         *  <li>Authority is null and path starts with "//"
337         * </ul>
338         * @return an URI instance, never <code>null</code>
339         */
340        public final static URI createURI(final String scheme, final String auth,
341            final String path, final String query, final String fragment) throws URIException
342        {
343            rand();
344            if (path==null) throw new URIException("Cannot pass a null path", null);
345            return instance.createURI(scheme, auth, path, query, fragment);
346        }
